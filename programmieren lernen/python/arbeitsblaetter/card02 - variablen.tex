meinWert = 5

# "meinWert" ausgeben
print(meinWert) # 5

# "meinWert" verdoppeln
meinWert = meinWert * 2
print(meinWert) # 10

# mit Variablen Rechnen:
zweiterWert = 2

# geht auch mit Zeichenketten:
meinWert = "Hallo"
zweiterWert = "Welt!"
print(meinWert) # Hallo
print(zweiterWert) # Welt!
print(meinWert + "..." + zweiterWert) # Hallo...Welt!

# Wert einer Variablen in einer anderen Variablen speichern:
hallo = meinWert
print(meinWert) # Hallo
print(hallo)  # Hallo
